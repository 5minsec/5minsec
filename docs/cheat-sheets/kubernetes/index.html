kubectl create deployment nginx-depl --image=nginx

# output

kubectl get deployment

# output

kubectl edit deployment nginx-depl

# output

# get logs
kubectl logs nginx-depl-7fc44fc5d4-45h68

# > pod container logs

kubectl delete deployment nginx-depl

Deploymentâ€‹

apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx-deployment
  labels:
    app: nginx
spec: # spec for deployment
  replicas: 1
  selector:
    matchLabels:
      app: nginx
  template:
    metadata:
      labels:
        app: nginx
    spec: # spec for a pod
      containers:
      - name: nginx
        image: nginx:1.16
        ports:
        - containerPort: 80
